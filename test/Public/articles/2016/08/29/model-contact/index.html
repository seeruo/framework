模型关联

<?php

namespace App;

use Illuminate\Database\Eloquent\Model;

class User extends Model
{
    /**
     * 获取与用户关联的电话号码记录。
     */
    public function phone()
    {
        return $this->hasOne('App\Phone', 'id', 'u_id');
    }
}

使用:

$phone = User::find(1)->phone;
